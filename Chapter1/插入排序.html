第4节：插入排序

import java.util.Arrays;
import java.util.Random;

public class InsertSort {
    public static void insertSort(int[] array) {
        int n = array.length;
        for (int i = 1; i < n - 1; i++) {
            int value = array[i];
            int j = i - 1;
            for (; j >= 0; j--) {
                if (array[j] > value) {
                    array[j + 1] = array[j];
                } else {
                    break;
                }
            }
            array[j + 1] = value;
        }
    }

    public static void main(String[] args) {
        int n = 20;
        int[] a = new int[n];
        Random r = new Random();
        for (int i = 0; i < n; i++) {
            a[i] = r.nextInt(100);
        }
        InsertSort.insertSort(a);
        System.out.println(Arrays.toString(a));
    }
}
